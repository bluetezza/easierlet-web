  <script>
    const SUPABASE_URL = 'https://ffzknvcptqjlkxmkdxuk.supabase.co'
    const SUPABASE_ANON_KEY = 'eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9.eyJpc3MiOiJzdXBhYmFzZSIsInJlZiI6ImZmemtudmNwdHFqbGt4bWtkeHVrIiwicm9sZSI6ImFub24iLCJpYXQiOjE3NzUyNTQ0ODQsImV4cCI6MjA5MDgzMDQ4NH0.YRdkBgoVm7cpUJlsK8T4nP6iNE3vgfOmODImiAkPhr4'

    const params = new URLSearchParams(window.location.search)
    const token = params.get('token')
    const app = document.getElementById('app')

    async function query(table, filters) {
      let url = `${SUPABASE_URL}/rest/v1/${table}?`
      const parts = Object.entries(filters).map(([k, v]) => `${k}=eq.${v}`)
      url += parts.join('&') + '&select=*'
      const res = await fetch(url, {
        headers: {
          'apikey': SUPABASE_ANON_KEY,
          'Authorization': `Bearer ${SUPABASE_ANON_KEY}`
        }
      })
      return res.json()
    }

    async function patch(table, id, data) {
      const res = await fetch(`${SUPABASE_URL}/rest/v1/${table}?id=eq.${id}`, {
        method: 'PATCH',
        headers: {
          'apikey': SUPABASE_ANON_KEY,
          'Authorization': `Bearer ${SUPABASE_ANON_KEY}`,
          'Content-Type': 'application/json',
          'Prefer': 'return=minimal'
        },
        body: JSON.stringify(data)
      })
      return res.ok
    }

    async function load() {
      if (!token) {
        app.innerHTML = '<div class="error"><h2>Invalid link</h2><p>Please contact your landlord for a new link.</p></div>'
        return
      }

      try {
        // Load report
        const reports = await query('inventory_reports', { access_token: token })
        if (!reports.length) {
          app.innerHTML = '<div class="error"><h2>Report not found</h2><p>This link may have expired. Please contact your landlord.</p></div>'
          return
        }
        const report = reports[0]

        // Load property
        const properties = await query('properties', { id: report.property_id })
        const property = properties[0]

        // Load rooms
        const rooms = await query('inventory_rooms', { inventory_id: report.id })
        rooms.sort((a, b) => a.order_index - b.order_index)

        // Load items for each room
        const allItems = {}
        for (const room of rooms) {
          const items = await query('inventory_items', { room_id: room.id })
          items.sort((a, b) => a.order_index - b.order_index)
          allItems[room.id] = items
        }

        renderForm(report, property, rooms, allItems)

      } catch (err) {
        app.innerHTML = `<div class="error"><h2>Something went wrong</h2><p>${err.message}</p></div>`
      }
    }

    function renderForm(report, property, rooms, allItems) {
      let html = `
        <div class="intro-card">
          <h1>Inventory for <span class="address">${property?.address ?? 'Your Property'}</span></h1>
          <p>Hello ${report.tenant_name ?? 'there'}, please review each room and item below and add your own observations. This helps protect both you and your landlord at the end of your tenancy.</p>
        </div>
        <form id="inventoryForm">
      `

      for (const room of rooms) {
        const items = allItems[room.id] ?? []
        html += `<div class="room"><h2 class="room-title">${room.name}</h2>`

        for (const item of items) {
          html += `
            <div class="item">
              <div class="item-header">
                <span class="item-name">${item.name}</span>
                ${item.landlord_condition ? `<span class="landlord-badge">Landlord: ${item.landlord_condition}</span>` : ''}
              </div>
              ${item.landlord_notes ? `<p class="landlord-notes">Landlord notes: ${item.landlord_notes}</p>` : ''}
              <div class="field-group">
                <label>Your assessment of condition</label>
                <select name="item_condition_${item.id}">
                  <option value="">Select condition</option>
                  ${['Excellent','Good','Fair','Poor','Damaged'].map(c => 
                    `<option value="${c}" ${item.tenant_condition === c ? 'selected' : ''}>${c}</option>`
                  ).join('')}
                </select>
                <label>Your notes (optional)</label>
                <textarea name="item_notes_${item.id}" placeholder="Any comments about this item...">${item.tenant_notes ?? ''}</textarea>
              </div>
            </div>
          `
        }

        html += `
          <div class="field-group">
            <label>Your overall notes for this room (optional)</label>
            <textarea name="room_notes_${room.id}" placeholder="Any general comments about this room...">${room.tenant_condition ?? ''}</textarea>
          </div>
          </div>
        `
      }

      html += `
        <div class="overall-notes">
          <h2>Your Overall Notes</h2>
          <div class="field-group">
            <label>Any additional comments about the property</label>
            <textarea name="tenant_notes" placeholder="Any general comments about the property condition...">${report.tenant_notes ?? ''}</textarea>
          </div>
        </div>
        <button type="submit" class="submit-btn" id="submitBtn">Submit My Review</button>
        </form>
      `

      app.innerHTML = html

      document.getElementById('inventoryForm').addEventListener('submit', async (e) => {
        e.preventDefault()
        const btn = document.getElementById('submitBtn')
        btn.textContent = 'Submitting...'
        btn.disabled = true

        const formData = new FormData(e.target)
        let success = true

        // Update each item
        for (const room of rooms) {
          const items = allItems[room.id] ?? []
          
          // Update room notes
          const roomNotes = formData.get(`room_notes_${room.id}`)
          await patch('inventory_rooms', room.id, { tenant_condition: roomNotes })

          // Update items
          for (const item of items) {
            const condition = formData.get(`item_condition_${item.id}`)
            const notes = formData.get(`item_notes_${item.id}`)
            const ok = await patch('inventory_items', item.id, {
              tenant_condition: condition || null,
              tenant_notes: notes || null
            })
            if (!ok) success = false
          }
        }

        // Update report
        await patch('inventory_reports', report.id, {
          tenant_notes: formData.get('tenant_notes'),
          status: 'completed',
          completed_at: new Date().toISOString()
        })

        if (success) {
          app.innerHTML = `
            <div class="success">
              <div style="font-size:48px; margin-bottom:16px;">✓</div>
              <h2>Thank you!</h2>
              <p>Your inventory review has been submitted successfully.<br>Your landlord will be notified.</p>
            </div>
          `
        } else {
          btn.textContent = 'Submit My Review'
          btn.disabled = false
          alert('Something went wrong. Please try again.')
        }
      })
    }

    load()
  </script>
